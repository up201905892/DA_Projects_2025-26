211 static void writeAssignmentSection(std::ostream& os,
212 const SolverResult& result,
213 const ProblemData& data) {
214 int totalDemand = static_cast<int>(data.numSubmissions()) *
215 data.config.minReviewsPerSubmission;
216
217 if (result.isComplete(totalDemand) || !result.assignments.empty()) {
218 // --- Section 1: Assignments by submission (Figure 2, part 1) ---
219 os << "#SubmissionId,ReviewerId,Match\n";
220 for (const auto& a : result.assignments) {
221 os << a.submissionId << ", "
222 << a.reviewerId << ", "
223 << a.matchedDomain << "\n";
224 }
225
226 // --- Section 2: Assignments by reviewer (Figure 2, part 2) ---
227 // Build reviewer → submissions map
228 std::map<EntityId, std::vector<std::pair<EntityId, DomainCode>>> byReviewer;
229 for (const auto& a : result.assignments) {
230 byReviewer[a.reviewerId].push_back({a.submissionId, a.matchedDomain});
231 }
232
233 os << "#ReviewerId,SubmissionId,Match\n";
234 for (auto& [revId, subs] : byReviewer) {
235 // Sort submissions within each reviewer
236 std::sort(subs.begin(), subs.end());
237 for (const auto& [subId, domain] : subs) {
238 os << revId << ", " << subId << ", " << domain << "\n";
239 }
240 }
241
242 // --- Section 3: Total ---
243 os << "#Total: " << result.assignments.size() << "\n";
244 }
245
246 // --- Section 4: Unfulfilled submissions (Figure 3) ---
247 if (!result.unfulfilled.empty()) {
248 os << "#SubmissionId,Domain,MissingReviews\n";
249 for (const auto& u : result.unfulfilled) {
250 os << u.submissionId << ", "
251 << u.domain << ", "
252 << u.missingReviews << "\n";
253 }
254 }
255 }

266 static void writeRiskSection(std::ostream& os,
267 const SolverResult& result,
268 int riskParam) {
269 os << "#Risk Analysis: " << riskParam << "\n";
270 if (result.riskyReviewers.empty()) {
271 os << "No risky reviewers found.\n";
272 } else {
273 for (size_t i = 0; i < result.riskyReviewers.size(); i++) {
274 if (i > 0) os << ", ";
275 os << result.riskyReviewers[i];
276 }
277 os << "\n";
278 }

286 static std::string matchModeToString(MatchMode mode) {
287 switch (mode) {
288 case MatchMode::SILENT: return "0 (Silent)";
