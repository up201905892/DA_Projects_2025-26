73 ProblemData parse(const std::string& filename) const {
74 std::ifstream file(filename);
75 if (!file.is_open()) {
76 throw std::runtime_error("Cannot open input file: " + filename);
77 }
78
79 ProblemData data;
80 State state = State::IDLE;
81 std::string line;
82 int lineNumber = 0;
83 std::vector<std::string> errors;
84
85 // Track which mandatory sections we've seen
86 bool seenSubmissions = false;
87 bool seenReviewers = false;
88 bool seenParameters = false;
89 bool seenControl = false;
90
91 while (std::getline(file, line)) {
92 lineNumber++;
93 line = trim(line);
94
95 // Skip empty lines
96 if (line.empty()) continue;
97
98 // --- State transition: detect section headers ---
99 if (line[0] == '#') {
100 std::string header = toLower(line);
101
102 if (header == "#submissions") {
103 if (seenSubmissions) {
104 errors.push_back("Line " + std::to_string(lineNumber) +
105 ": Duplicate #Submissions section.");
106 }
107 state = State::SUBMISSIONS;
108 seenSubmissions = true;
109 } else if (header == "#reviewers") {
110 if (seenReviewers) {
111 errors.push_back("Line " + std::to_string(lineNumber) +
112 ": Duplicate #Reviewers section.");
113 }
114 state = State::REVIEWERS;
115 seenReviewers = true;
116 } else if (header == "#parameters") {
117 if (seenParameters) {
118 errors.push_back("Line " + std::to_string(lineNumber) +
119 ": Duplicate #Parameters section.");
120 }
121 state = State::PARAMETERS;
122 seenParameters = true;
123 } else if (header == "#control") {
124 if (seenControl) {
125 errors.push_back("Line " + std::to_string(lineNumber) +
126 ": Duplicate #Control section.");
127 }
128 state = State::CONTROL;
129 seenControl = true;
130 } else {
131 // Unknown section header or column-header comment — skip silently
132 // Do NOT change state: e.g. "#Id, Title, ..." is a column header
133 // inside a section and must not reset the current parsing state.
134 }
135 continue; // Header line consumed, move to next line
136 }
137
138 // --- Data lines: dispatch to section-specific parser ---
139 try {
140 switch (state) {
141 case State::SUBMISSIONS:
142 parseSubmissionLine(line, lineNumber, data);
143 break;
144 case State::REVIEWERS:
145 parseReviewerLine(line, lineNumber, data);
146 break;
147 case State::PARAMETERS:
148 parseParameterLine(line, lineNumber, data.config);
149 break;
150 case State::CONTROL:
151 parseControlLine(line, lineNumber, data.config);
152 break;
153 case State::IDLE:
154 // Data before any section header — ignore with warning
155 std::cerr << "Warning: line " << lineNumber
156 << " is outside any section, skipping." << std::endl;
157 break;
158 }
159 } catch (const ParseError& e) {
160 errors.push_back(e.what());
161 }
162 }
163
164 // --- Post-parse validation ---
165 if (!seenSubmissions) errors.push_back("Missing mandatory section: #Submissions");
166 if (!seenReviewers) errors.push_back("Missing mandatory section: #Reviewers");
167 if (!seenParameters) errors.push_back("Missing mandatory section: #Parameters");
168 if (!seenControl) errors.push_back("Missing mandatory section: #Control");
169
170 if (data.submissions.empty() && seenSubmissions) {
171 errors.push_back("#Submissions section is empty — no submissions to assign.");
172 }
173 if (data.reviewers.empty() && seenReviewers) {
174 errors.push_back("#Reviewers section is empty — no reviewers available.");
175 }
176
177 // Validate parameter consistency
178 if (data.config.minReviewsPerSubmission < 1) {
179 errors.push_back("MinReviewsPerSubmission must be >= 1, got: " +
180 std::to_string(data.config.minReviewsPerSubmission));
181 }
182 if (data.config.maxReviewsPerReviewer < 1) {
183 errors.push_back("MaxReviewsPerReviewer must be >= 1, got: " +
184 std::to_string(data.config.maxReviewsPerReviewer));
185 }
186 if (data.config.riskAnalysis < 0) {
187 errors.push_back("RiskAnalysis must be >= 0, got: " +
188 std::to_string(data.config.riskAnalysis));
189 }
190
191 // Report all collected errors
192 if (!errors.empty()) {
193 std::ostringstream oss;
194 oss << "Input file has " << errors.size() << " error(s):\n";
195 for (size_t i = 0; i < errors.size(); i++) {
196 oss << " [" << (i + 1) << "] " << errors[i] << "\n";
197 }
198 throw ParseError(oss.str(), 0, "validation");
199 }
200
201 return data;
202 }

257 static void parseSubmissionLine(const std::string& line, int lineNum,
258 ProblemData& data) {
259 auto tokens = splitCSV(line);
260
261 Submission sub;
262
263 // Full format: Id, Title, Authors, E-mail, Primary [, Secondary] → 5 or 6 tokens
264 // Legacy format: Id, Primary [, Secondary] → 2 or 3 tokens
265 if (tokens.size() == 5 || tokens.size() == 6) {
266 sub.id = parseIntField(tokens[0], "submission ID", lineNum, "#Submissions");
267 sub.title = tokens[1];
268 sub.authors = tokens[2];
269 sub.email = tokens[3];
270 sub.primaryDomain = parseIntField(tokens[4], "primary domain", lineNum, "#Submissions");
271 if (tokens.size() == 6) {
272 sub.secondaryDomain = parseIntField(tokens[5], "secondary domain",
273 lineNum, "#Submissions");
274 }
275 } else if (tokens.size() == 2 || tokens.size() == 3) {
276 // Legacy / test format: Id, Primary [, Secondary]
277 sub.id = parseIntField(tokens[0], "submission ID", lineNum, "#Submissions");
278 sub.primaryDomain = parseIntField(tokens[1], "primary domain", lineNum, "#Submissions");
279 if (tokens.size() == 3) {
280 sub.secondaryDomain = parseIntField(tokens[2], "secondary domain",
281 lineNum, "#Submissions");
282 }
283 } else {
284 throw ParseError(
285 "Submission line expects 5–6 fields (Id, Title, Authors, E-mail, Primary"
286 " [, Secondary]) or 2–3 legacy fields, got " +
287 std::to_string(tokens.size()) + ": \"" + line + "\"",
288 lineNum, "#Submissions");
289 }
290
291 // Validate: ID must be positive
292 if (sub.id <= 0) {
293 throw ParseError("Submission ID must be positive, got: " +
294 std::to_string(sub.id), lineNum, "#Submissions");
295 }
296 // Validate: domain codes must be positive
297 if (sub.primaryDomain <= 0) {
298 throw ParseError("Primary domain must be positive, got: " +
299 std::to_string(sub.primaryDomain), lineNum, "#Submissions");
300 }
301 if (sub.secondaryDomain.has_value() && sub.secondaryDomain.value() <= 0) {
302 throw ParseError("Secondary domain must be positive, got: " +
303 std::to_string(sub.secondaryDomain.value()),
304 lineNum, "#Submissions");
305 }
306
307 try {
308 data.addSubmission(sub);
309 } catch (const std::runtime_error& e) {
310 throw ParseError(e.what(), lineNum, "#Submissions");
311 }
312 }

406 static void parseParameterLine(const std::string& line, int lineNum,
407 Config& config) {
408 auto tokens = splitCSV(line);
409 if (tokens.size() != 2) {
410 throw ParseError(
411 "Parameter line expects 2 fields (key, value), got " +
412 std::to_string(tokens.size()) + ": \"" + line + "\"",
413 lineNum, "#Parameters");
414 }
415
416 std::string key = toLower(tokens[0]);
417
418 if (key == "minreviewspersubmission") {
419 config.minReviewsPerSubmission =
420 parseIntField(tokens[1], "MinReviewsPerSubmission", lineNum, "#Parameters");
421 } else if (key == "maxreviewsperreviewer") {
422 config.maxReviewsPerReviewer =
423 parseIntField(tokens[1], "MaxReviewsPerReviewer", lineNum, "#Parameters");
424 } else if (key == "primaryreviewerexpertise") {
425 config.primaryReviewerExpertise =
426 parseBoolField(tokens[1], "PrimaryReviewerExpertise", lineNum, "#Parameters");
427 } else if (key == "secondaryreviewerexpertise") {
428 config.secondaryReviewerExpertise =
429 parseBoolField(tokens[1], "SecondaryReviewerExpertise", lineNum, "#Parameters");
430 } else if (key == "primarysubmissiondomain") {
431 config.primarySubmissionDomain =
432 parseBoolField(tokens[1], "PrimarySubmissionDomain", lineNum, "#Parameters");
433 } else if (key == "secondarysubmissiondomain") {
434 config.secondarySubmissionDomain =
435 parseBoolField(tokens[1], "SecondarySubmissionDomain", lineNum, "#Parameters");
436 } else {
437 throw ParseError("Unrecognized parameter key: \"" + tokens[0] + "\"",
438 lineNum, "#Parameters");

458 static void parseControlLine(const std::string& line, int lineNum,
459 Config& config) {
460 auto tokens = splitCSV(line);
461 if (tokens.size() != 2) {
462 throw ParseError(
463 "Control line expects 2 fields (key, value), got " +
464 std::to_string(tokens.size()) + ": \"" + line + "\"",
465 lineNum, "#Control");
466 }
467
468 std::string key = toLower(tokens[0]);
469
470 if (key == "generateassignments") {
471 int val = parseIntField(tokens[1], "GenerateAssignments", lineNum, "#Control");
472 try {
473 config.matchMode = toMatchMode(val);
474 } catch (const std::invalid_argument& e) {
475 throw ParseError(e.what(), lineNum, "#Control");
476 }
477 } else if (key == "riskanalysis") {
478 config.riskAnalysis =
479 parseIntField(tokens[1], "RiskAnalysis", lineNum, "#Control");
480 } else if (key == "outputfilename") {
481 config.outputFileName = stripQuotes(tokens[1]);
482 if (config.outputFileName.empty()) {
483 throw ParseError("OutputFileName cannot be empty.", lineNum, "#Control");
484 }
485 } else {
486 throw ParseError("Unrecognized control key: \"" + tokens[0] + "\"",
487 lineNum, "#Control");
488 }

517 static std::vector<std::string> splitCSV(const std::string& line) {
518 std::vector<std::string> tokens;
519 std::string current;
520 bool inQuotes = false;
521
522 for (size_t i = 0; i < line.size(); ++i) {
523 char c = line[i];
524 if (c == '"') {
525 inQuotes = !inQuotes; // toggle quote mode; quote char itself not stored
526 } else if (c == ',' && !inQuotes) {
527 std::string t = trim(current);
528 if (!t.empty()) {
529 tokens.push_back(t);
530 }
531 current.clear();
532 } else {
533 current += c;
534 }
535 }
536
537 // Flush the last field
538 std::string t = trim(current);
539 if (!t.empty()) {
540 tokens.push_back(t);
541 }
542
543 return tokens;
544 }

555 static std::string trim(const std::string& s) {
556 auto start = s.find_first_not_of(" \t\r\n");
557 if (start == std::string::npos) return "";
558 auto end = s.find_last_not_of(" \t\r\n");
559 return s.substr(start, end - start + 1);
560 }

571 static std::string toLower(const std::string& s) {
572 std::string result = s;
573 std::transform(result.begin(), result.end(), result.begin(),
574 [](unsigned char c) { return std::tolower(c); });
575 return result;
576 }

587 static std::string stripQuotes(const std::string& s) {
588 std::string result = trim(s);
589 if (result.size() >= 2 && result.front() == '"' && result.back() == '"') {
590 result = result.substr(1, result.size() - 2);
591 }
592 return result;
593 }

608 static int parseIntField(const std::string& token, const std::string& fieldName,
609 int lineNum, const std::string& section) {
610 try {
611 size_t pos = 0;
612 int value = std::stoi(token, &pos);
613 // Ensure the entire token was consumed (no trailing garbage)
614 if (pos != token.size()) {
615 throw ParseError(
616 fieldName + " has trailing characters: \"" + token + "\"",
617 lineNum, section);
618 }
619 return value;
620 } catch (const std::invalid_argument&) {
621 throw ParseError(
622 fieldName + " is not a valid integer: \"" + token + "\"",
623 lineNum, section);
624 } catch (const std::out_of_range&) {
625 throw ParseError(
626 fieldName + " is out of range: \"" + token + "\"",
627 lineNum, section);
628 }
629 }

644 static bool parseBoolField(const std::string& token, const std::string& fieldName,
645 int lineNum, const std::string& section) {
646 int value = parseIntField(token, fieldName, lineNum, section);
647 if (value != 0 && value != 1) {
648 throw ParseError(
649 fieldName + " must be 0 or 1, got: " + std::to_string(value),
650 lineNum, section);
651 }
652 return value == 1;
653 }
