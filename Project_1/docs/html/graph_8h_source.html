143 void addEdge(int u, int v, int cap) {
144 assert(u >= 0 && u < numNodes());
145 assert(v >= 0 && v < numNodes());
146 assert(cap >= 0);
147
148 // Forward edge: u → v with capacity cap
149 Edge forward{v, cap, 0, static_cast<int>(adj_[v].size())};
150 // Reverse edge: v → u with capacity 0
151 Edge backward{u, 0, 0, static_cast<int>(adj_[u].size())};
152
153 adj_[u].push_back(forward);

182 int edmondsKarp(int source, int sink) {
183 int totalFlow = 0;
184
185 // parent[v] = {node_u, edge_index_in_adj[u]} that leads to v in BFS
186 std::vector<std::pair<int, int>> parent(numNodes());
187
188 while (true) {
189 // --- BFS to find shortest augmenting path ---
190 int bottleneck = bfs(source, sink, parent);
191 if (bottleneck == 0) break; // No augmenting path found
192
193 // --- Augment flow along the path ---
194 totalFlow += bottleneck;
195 int v = sink;
196 while (v != source) {
197 auto [u, edgeIdx] = parent[v];
198 adj_[u][edgeIdx].flow += bottleneck;
199 adj_[v][adj_[u][edgeIdx].rev].flow -= bottleneck;
200 v = u;
201 }
202 }
203
204 return totalFlow;
205 }

289 int bfs(int source, int sink, std::vector<std::pair<int, int>>& parent) {
290 std::fill(parent.begin(), parent.end(), std::make_pair(-1, -1));
291 parent[source] = {source, -1};
292
293 // BFS queue stores {node, bottleneck_so_far}
294 std::queue<std::pair<int, int>> q;
295 q.push({source, INT_MAX});
296
297 while (!q.empty()) {
298 auto [u, pathFlow] = q.front();
299 q.pop();
300
301 for (int i = 0; i < static_cast<int>(adj_[u].size()); i++) {
302 Edge& e = adj_[u][i];
303 // Only traverse edges with positive residual capacity
304 // and unvisited destinations
305 if (parent[e.dest].first == -1 && e.residual() > 0) {
306 parent[e.dest] = {u, i};
307 int newFlow = std::min(pathFlow, e.residual());
308
309 if (e.dest == sink) {
310 return newFlow; // Found augmenting path
311 }
312 q.push({e.dest, newFlow});
313 }
314 }
315 }
316
317 return 0; // No augmenting path to sink
318 }

365 static Graph buildNetwork(const ProblemData& data) {
366 int m = static_cast<int>(data.numReviewers());
367 int n = static_cast<int>(data.numSubmissions());
368 int totalNodes = 2 + m + n; // Source + Sink + Reviewers + Submissions
369
370 Graph g(totalNodes);
371
372 const Config& cfg = data.config;
373 bool useSubSecondary = cfg.useSubmissionSecondary();
374 bool useRevSecondary = cfg.useReviewerSecondary();
375
376 // --- Layer 1: Source → Reviewers ---
377 for (int j = 0; j < m; j++) {
378 int revNode = reviewerNode(j);
379 g.addEdge(SOURCE, revNode, cfg.maxReviewsPerReviewer);
380 }
381
382 // --- Layer 2: Reviewers → Submissions (compatibility edges) ---
383 for (int j = 0; j < m; j++) {
384 const Reviewer& rev = data.reviewers[j];
385 int revNode = reviewerNode(j);
386
387 for (int i = 0; i < n; i++) {
388 const Submission& sub = data.submissions[i];
389
390 if (isCompatible(rev, sub, useRevSecondary, useSubSecondary)) {
391 int subNode = submissionNode(i, m);
392 g.addEdge(revNode, subNode, 1);
393 }
394 }
395 }
396
397 // --- Layer 3: Submissions → Sink ---
398 for (int i = 0; i < n; i++) {
399 int subNode = submissionNode(i, m);
400 g.addEdge(subNode, SINK, cfg.minReviewsPerSubmission);
401 }
402
403 return g;
404 }
405

470 static bool isCompatible(const Reviewer& rev, const Submission& sub,
471 bool useRevSec, bool useSubSec) {
472 // Check primary expertise against submission domains

544 static SolverResult solveWithRiskAnalysis(const ProblemData& data) {
545 int m = static_cast<int>(data.numReviewers());
546 int n = static_cast<int>(data.numSubmissions());
547 int requiredFlow = n * data.config.minReviewsPerSubmission;
548
549 // First: solve the base problem
550 Graph g = NetworkBuilder::buildNetwork(data);
551 int baseFlow = g.edmondsKarp(NetworkBuilder::SOURCE, NetworkBuilder::SINK);
552
553 SolverResult result = extractResult(g, data, m, n, baseFlow);
554
555 // If base assignment already fails, no point in risk analysis
556 if (baseFlow < requiredFlow) {
557 return result;
558 }
559
560 // Save original graph for restoration
561 Graph original = NetworkBuilder::buildNetwork(data);
562
563 // Test each reviewer's removal
564 for (int j = 0; j < m; j++) {
565 // Rebuild fresh graph for each test

578 }

604 static SolverResult extractResult(const Graph& g, const ProblemData& data,
605 int m, int n, int maxFlow) {
606 SolverResult result;
607 result.maxFlow = maxFlow;
608
609 // Track how many reviews each submission received
610 std::vector<int> reviewCount(n, 0);
611
612 // Scan all reviewer nodes for edges with positive flow
613 for (int j = 0; j < m; j++) {
614 int revNode = NetworkBuilder::reviewerNode(j);
615 const auto& edges = g.getAdj(revNode);
616
617 for (const Edge& e : edges) {
618 // Only forward edges to submission nodes with flow > 0
619 if (e.flow <= 0 || e.capacity == 0) continue;
620
621 int subIdx = NetworkBuilder::nodeToSubmissionIdx(e.dest, m, n);
622 if (subIdx < 0) continue; // Not a submission node (e.g., reverse to Source)
623
624 // Determine the matched domain
625 DomainCode matchedDomain = findMatchedDomain(
626 data.reviewers[j], data.submissions[subIdx], data.config);
627
628 Assignment asgn;
629 asgn.submissionId = data.submissions[subIdx].id;
630 asgn.reviewerId = data.reviewers[j].id;
631 asgn.matchedDomain = matchedDomain;
632 result.assignments.push_back(asgn);
633
634 reviewCount[subIdx]++;
635 }
636 }
637
638 // Identify unfulfilled submissions
639 for (int i = 0; i < n; i++) {
640 int missing = data.config.minReviewsPerSubmission - reviewCount[i];
641 if (missing > 0) {
642 UnfulfilledSubmission u;
643 u.submissionId = data.submissions[i].id;
644 u.domain = data.submissions[i].primaryDomain;
645 u.missingReviews = missing;
646 result.unfulfilled.push_back(u);
647 }
648 }
649
650 // Sort for deterministic output
651 std::sort(result.assignments.begin(), result.assignments.end());
652 std::sort(result.unfulfilled.begin(), result.unfulfilled.end());
653
654 return result;
655 }

671 static DomainCode findMatchedDomain(const Reviewer& rev, const Submission& sub,
672 const Config& cfg) {
673 // Priority 1: Primary expertise matches primary domain
674 if (rev.primaryExpertise == sub.primaryDomain) {
675 return sub.primaryDomain;
676 }
677
678 // Priority 2: Primary expertise matches secondary domain
679 if (cfg.useSubmissionSecondary() &&
680 sub.secondaryDomain.has_value() &&
681 rev.primaryExpertise == sub.secondaryDomain.value()) {
682 return sub.secondaryDomain.value();
683 }
684
685 // Priority 3: Secondary expertise matches primary domain
686 if (cfg.useReviewerSecondary() &&
687 rev.secondaryExpertise.has_value() &&
688 rev.secondaryExpertise.value() == sub.primaryDomain) {
689 return sub.primaryDomain;
690 }
691
692 // Priority 4: Secondary expertise matches secondary domain
693 if (cfg.useReviewerSecondary() && cfg.useSubmissionSecondary() &&
694 rev.secondaryExpertise.has_value() &&
695 sub.secondaryDomain.has_value() &&
696 rev.secondaryExpertise.value() == sub.secondaryDomain.value()) {
697 return sub.secondaryDomain.value();
698 }
699
700 // Fallback (should not happen if the edge exists)
701 return sub.primaryDomain;
702 }
